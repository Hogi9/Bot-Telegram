git init
git status (untuk cek apakah ada progress yang belum tercommit)
git add <file> (untuk menambahkan file di repository) bisa gunakan git add .
git commit -m "Text" (-m untuk tambah message)
git commit -am "Text" (-a untuk git add dan -m untuk message)

git branch <nama_branch>
git branch --merged (untuk mengecek branch yang sudah di merger)
git branch -d <nama_branch> (untuk menghapus branch yang sudah di merge)
git branch -D <nama_branch> (untuk menghapus branch yang belum di merge)
git merge <nama_branch> (posisi head atau pointer ada di master)
git checkout <hash> -- <file> (untuk kembali ke checkpoint pada file tertentu)
git checkout <nama_branch> 

git push
git pull
git fetch -> git pull (digunakan jika terdapat conflict antara remote dengan local)

git config
git log (untuk melihat log commit)
git log --all --decorate --oneline --graph (untuk melihat log dengan grafik)
alias graph="git log --all --decorate --oneline --graph" (untuk membuat alias)
